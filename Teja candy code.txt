#include <stdio.h>

void main()
{
    int tea,candy;
    printf("Enter the rating of tea and candy\n");
    scanf("%d %d",&tea,&candy);
    if(tea<5||candy<5)
    {
        printf("0");
    }
    else if(tea<=10||candy<=10)
    {
        printf("2");
    }
    else
    {
        printf("1");
    }
}
